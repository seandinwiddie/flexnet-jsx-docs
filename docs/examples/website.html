Website Example

Building a complete website with FlexNet JSX

Overview

This example demonstrates how to build a complete multi-page website using FlexNet JSX, showcasing routing, data fetching, state management, and more.

Key Features

  • Functional Routing System with route parameters and guards
  • API Integration with the Effect system
  • Component Composition with Header, Footer, and Layout components
  • Form Handling with validation using Maybe and Either

Key Components

Router Implementation

// src/router/router.js
import { createStore } from '../systems/state/store.js'; // A real app would import this

// --- Action Types and Reducer ---
const SET_ROUTE = 'SET_ROUTE';

const routerReducer = (state, action) => {
    switch(action.type) {
        case SET_ROUTE:
            return { ...state, ...action.payload };
        default:
            return state;
    }
};

// --- Route Matching Logic ---
const matchRoute = (routes, path) => {
    for (const route of routes) {
        // Creates a regex from the route path (e.g., /blog/:id)
        const pattern = new RegExp(`^${route.path.replace(/:\w+/g, '([^\\/]+)')}$`);
        const match = path.match(pattern);
        
        if (match) {
            // Extracts param names (e.g., ['id'])
            const paramNames = (route.path.match(/:\w+/g) || []).map(p => p.substring(1));
            // Creates a params object (e.g., { id: '123' })
            const params = paramNames.reduce((acc, name, i) => ({ ...acc, [name]: match[i + 1] }), {});
            return { component: route.component, params };
        }
    }
    // Fallback to the not-found route
    const notFoundRoute = routes.find(r => r.path === '*');
    return { component: notFoundRoute ? notFoundRoute.component : () => createElement('div', null, '404 Not Found'), params: {} };
};

// --- Router Creation ---
export const createRouter = (routes) => {
    const initialState = {
        path: window.location.pathname,
        component: null,
        params: {}
    };
    const store = createStore(routerReducer, initialState);

    const navigate = (path) => {
        // Side effect: update browser history
        window.history.pushState({}, '', path);
        
        const { component, params } = matchRoute(routes, path);
        
        // Dispatch action to update the router's state
        store.dispatch({
            type: SET_ROUTE,
            payload: { path, component, params }
        });
    };

    const init = () => {
        // Listen for back/forward navigation
        window.addEventListener('popstate', () => {
            navigate(window.location.pathname);
        });

        // Intercept clicks on internal links
        document.addEventListener('click', e => {
            const link = e.target.closest('a[data-internal]');
            if (link) {
                e.preventDefault();
                navigate(link.getAttribute('href'));
            }
        });

        // Initial navigation to set the component for the current URL
        navigate(window.location.pathname);
    };

    return {
        init,
        navigate,
        subscribe: store.subscribe,
        getState: store.getState
    };
};

API Client with Effects

// src/api/client.js
import { Effect } from '../systems/effects/functions.js'; // Assumed from api-reference.md
import { Either } from '../core/types/either.js'; // Assumed from api-reference.md

// NOTE: The Effect and Either types would typically be imported from the framework core.
// They are redefined here for the sake of a self-contained example.

const Effect = {
    of: value => ({ type: 'Effect', run: () => value }),
    map: fn => effect => Effect.of(() => fn(effect.run())),
    chain: fn => effect => Effect.of(() => fn(effect.run()).run()),
    run: effect => effect.run()
};

const Either = {
    Left: value => ({ type: 'Left', value, isLeft: true, isRight: false, fold: (f, g) => f(value) }),
    Right: value => ({ type: 'Right', value, isLeft: false, isRight: true, fold: (f, g) => g(value) })
};

export const apiClient = {
  fetch: (url, options = {}) => Effect.of(() => 
    fetch(url, {
        ...options,
        headers: { 'Content-Type': 'application/json', ...options.headers }
    }).then(res => {
        if (!res.ok) {
            return res.json()
                .catch(() => ({ message: res.statusText })) // Handle non-JSON error bodies
                .then(body => Promise.reject(Either.Left({ status: res.status, body })));
        }
        return res.json().then(data => Either.Right(data));
    }).catch(networkError => {
        // This catches fetch failures (e.g., network down) and rejected promises from above
        const error = networkError.isLeft ? networkError : Either.Left({ status: 'NETWORK_ERROR', body: networkError.message });
        return error;
    })
  ),
  
  get: (url) => apiClient.fetch(url),
  post: (url, data) => apiClient.fetch(url, { method: 'POST', body: JSON.stringify(data) }),
  put: (url, data) => apiClient.fetch(url, { method: 'PUT', body: JSON.stringify(data) }),
  delete: (url) => apiClient.fetch(url, { method: 'DELETE' })
};

Form Handling with Validation

// src/features/contact/ContactForm.js
import { createElement } from '../../core/jsx.js';
import { createStore } from '../../systems/state/store.js';
import { Either } from '../../core/types/either.js';
import { apiClient } from '../../api/client.js';
import { Effect } from '../../systems/effects/effect.js';
import { sanitize, validateEmail, validateRequired } from '../../utils/validators.js';

// NOTE: These utilities would be in separate files in a real project.
// They are included here for a self-contained example.

const validateRequired = (value, errorMsg = 'This field is required') => 
    value && value.trim().length > 0 ? Either.Right(value) : Either.Left(errorMsg);

const validateEmail = (email, errorMsg = 'Invalid email address') => {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(email) ? Either.Right(email) : Either.Left(errorMsg);
};

const sanitize = input => {
    const div = document.createElement('div');
    div.textContent = input;
    return div.innerHTML;
};

// --- Form Reducer & Actions ---
const formAction = {
    UPDATE_FIELD: 'UPDATE_FIELD', SET_ERRORS: 'SET_ERRORS',
    SUBMIT_START: 'SUBMIT_START', SUBMIT_SUCCESS: 'SUBMIT_SUCCESS',
    SUBMIT_FAILURE: 'SUBMIT_FAILURE', RESET_FORM: 'RESET_FORM'
};

const formInitialState = {
    name: '', email: '', message: '',
    errors: {}, submitting: false, submitted: false
};

const formReducer = (state, action) => {
    switch (action.type) {
        case formAction.UPDATE_FIELD:
            return {
                ...state,
                [action.payload.field]: action.payload.value,
                errors: { ...state.errors, [action.payload.field]: null, form: null }
            };
        case formAction.SET_ERRORS:
            return { ...state, errors: action.payload };
        case formAction.SUBMIT_START:
            return { ...state, submitting: true, errors: { ...state.errors, form: null } };
        case formAction.SUBMIT_SUCCESS:
            return { ...formInitialState, submitted: true };
        case formAction.SUBMIT_FAILURE:
            return { ...state, submitting: false, errors: { ...state.errors, form: action.payload.error } };
        case formAction.RESET_FORM:
            return { ...formInitialState };
        default:
            return state;
    }
};

// --- ContactForm Component ---
export const ContactForm = () => {
    const formStore = createStore(formReducer, formInitialState);
    const { getState, dispatch, subscribe } = formStore;

    // This is a simplified way to handle component re-rendering for this example.
    const formElement = document.createElement('div');
    const renderComponent = () => {
        const state = getState();
        const newVNode = renderLogic(state); // Get the virtual DOM
        render(newVNode, formElement); // Render it
    };
    subscribe(renderComponent); // Re-render on state changes
    
    const validate = () => {
        const { name, email, message } = getState();
        const errors = {};
        validateRequired(name).fold(e => errors.name = e, () => {});
        validateEmail(email).fold(e => errors.email = e, () => {});
        validateRequired(message).fold(e => errors.message = e, () => {});
        dispatch({ type: formAction.SET_ERRORS, payload: errors });
        return Object.keys(errors).length === 0;
    };
  
    const handleSubmit = (e) => {
        e.preventDefault();
        if (!validate()) return;
        
        dispatch({ type: formAction.SUBMIT_START });
        const { name, email, message } = getState();
        
        const apiCallEffect = apiClient.post('/api/contact', {
            name: sanitize(name),
            email: sanitize(email),
            message: sanitize(message)
        });

        Effect.run(apiCallEffect).then(eitherResult => {
            eitherResult.fold(
                error => dispatch({ type: formAction.SUBMIT_FAILURE, payload: { error: error.body.message || 'Submission failed' } }),
                () => dispatch({ type: formAction.SUBMIT_SUCCESS })
            );
        });
    };
  
    const renderLogic = ({ name, email, message, errors, submitting, submitted }) => {
        if (submitted) {
            return createElement('div', { className: 'text-center p-4 bg-green-100 border border-green-300 rounded' },
                createElement('h3', { className: 'font-bold text-lg' }, 'Thank you for your message!'),
                createElement('button', {
                    className: 'mt-4 px-4 py-2 bg-primary text-white rounded hover:bg-blue-700',
                    onClick: () => dispatch({ type: formAction.RESET_FORM })
                }, 'Send another message')
            );
        }

        return createElement('form', { onSubmit: handleSubmit, className: 'space-y-4' },
            createElement('div', null,
                createElement('label', { htmlFor: 'name', className: 'block font-semibold' }, 'Name'),
                createElement('input', {
                    type: 'text', id: 'name', value: name,
                    onInput: (e) => dispatch({ type: formAction.UPDATE_FIELD, payload: { field: 'name', value: e.target.value } }),
                    className: `w-full p-2 border rounded ${errors.name ? 'border-red-500' : 'border-gray-300'}`
                }),
                errors.name && createElement('div', { className: 'text-red-500 text-sm mt-1' }, errors.name)
            ),
            createElement('div', null,
                createElement('label', { htmlFor: 'email', className: 'block font-semibold' }, 'Email'),
                createElement('input', {
                    type: 'email', id: 'email', value: email,
                    onInput: (e) => dispatch({ type: formAction.UPDATE_FIELD, payload: { field: 'email', value: e.target.value } }),
                    className: `w-full p-2 border rounded ${errors.email ? 'border-red-500' : 'border-gray-300'}`
                }),
                errors.email && createElement('div', { className: 'text-red-500 text-sm mt-1' }, errors.email)
            ),
            createElement('div', null,
                createElement('label', { htmlFor: 'message', className: 'block font-semibold' }, 'Message'),
                createElement('textarea', {
                    id: 'message', value: message, rows: 5,
                    onInput: (e) => dispatch({ type: formAction.UPDATE_FIELD, payload: { field: 'message', value: e.target.value } }),
                    className: `w-full p-2 border rounded ${errors.message ? 'border-red-500' : 'border-gray-300'}`
                }),
                errors.message && createElement('div', { className: 'text-red-500 text-sm mt-1' }, errors.message)
            ),
            errors.form && createElement('div', { className: 'p-2 bg-red-100 text-red-700 border border-red-300 rounded' }, errors.form),
            createElement('button', {
                type: 'submit',
                className: 'w-full px-4 py-2 bg-primary text-white font-bold rounded hover:bg-blue-700 disabled:bg-gray-400',
                disabled: submitting
            }, submitting ? 'Sending...' : 'Send Message')
        );
    };

    renderComponent(); // Initial render
    return formElement; // Return the DOM element that the component manages
};

                    
Main Application

                    
// src/index.js
import { createElement, render } from './systems/core.js'; // Assuming a consolidated core file
import { createRouter } from './router/router.js';
import { Layout } from './components/Layout.js';
import { HomePage } from './pages/HomePage.js';
import { AboutPage } from './pages/AboutPage.js';
import { BlogPage } from './pages/BlogPage.js';
import { BlogPostPage } from './pages/BlogPostPage.js';
import { ContactPage } from './pages/ContactPage.js';
import { NotFoundPage } from './pages/NotFoundPage.js';

// Define routes
const routes = [
  { path: '/', component: HomePage },
  { path: '/about', component: AboutPage },
  { path: '/blog', component: BlogPage },
  { path: '/blog/:id', component: BlogPostPage },
  { path: '/contact', component: ContactPage },
  { path: '*', component: NotFoundPage }
];

// App component that initializes the router and subscribes to changes
const App = (rootElement) => {
  const router = createRouter(routes);

  router.subscribe((routeState) => {
    const { component, params } = routeState;
    
    // Ensure we have a component to render
    if (!component) return;

    // Render the current route component within the main layout
    render(
      createElement(Layout, null,
        createElement(component, { params })
      ),
      rootElement
    );
  });

  // Initialize the router to handle the initial page load and link clicks
  router.init();
};

// Start the application after the DOM is loaded
document.addEventListener('DOMContentLoaded', () => {
    const root = document.getElementById('root');
    if (root) {
        App(root);
    }
});

Side Effect Management

                            
Effects are cleanly isolated and managed, making the code easier to test and maintain.
