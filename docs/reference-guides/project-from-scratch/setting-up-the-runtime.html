4 Setting Up the Runtime

Implement the core FlexNet runtime. This includes a factory function and a transform to bootstrap the application.

Create core/runtime/factory.js:

// FlexNet factory
export const createElement = (type, props, ...children) => ({
    type,
    props: { ...props, children: children.flat() }
});

Create core/runtime/transform.js:

// FlexNet transform runtime
import { render } from '../../systems/render/functions.js';
import { App } from '../../features/counter/index.js';

// Bootstrap application
document.addEventListener('DOMContentLoaded', () => {
  console.log('FlexNet transform runtime initialized');
  const appContainer = document.getElementById('app');
  if (appContainer) {
    render(App, appContainer);
  }
});
